### 4. Integrate mailauth in `src/server/mx-server.ts`

```ts
import { authenticate as mailAuth } from 'mailauth'
```

In `onData` after raw buffer collection:
```ts
const authResult = await mailAuth(raw, {
    ip: session.remoteAddress || '0.0.0.0',
    helo: session.hostNameAppearsAs || 'unknown',
    mta: process.env.MAIL_HOST || 'mail.skale.club',
    sender: session.envelope.mailFrom?.address || '',
}).catch((err: Error) => {
    console.error('[MX] mailauth error:', err.message)
    return null
})

let sealedRaw = raw
let isSpam = false

if (authResult) {
    // Prepend Authentication-Results header
    sealedRaw = Buffer.concat([
        Buffer.from(authResult.headers + '\r\n'),
        raw,
    ])

    const dmarc = authResult.dmarc
    if (dmarc?.status?.result === 'fail') {
        if (dmarc.policy === 'reject') {
            return callback(new Error('550 5.7.1 DMARC policy violation'))
        }
        if (dmarc.policy === 'quarantine') {
            isSpam = true
        }
    }
}

const parsed = await parseRawEmail(sealedRaw)
// ... deliver to inbox OR spam folder if isSpam
```

Adjust `storeInbound` signature to accept `isSpam` flag and route to `spam` folder when true:

const folder = await db.query.mailFolders.findFirst({

- Private key format mismatch â€” some systems store PKCS#1, others PKCS#8. mailauth/nodemailer expect PEM with headers. If DB stores raw base64, wrap with `-----BEGIN PRIVATE KEY-----` when loading.
- `mailauth` does DNS lookups per message â€” slow in high traffic. Cache positive results by sender IP + selector (future optimization).
- False positives on forwarded mail (ARC needed eventually)
