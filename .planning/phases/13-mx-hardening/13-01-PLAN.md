### 2. Create `src/server/lib/mx-guard.ts`
```ts
import { promises as dns } from 'dns'

// Per-IP connection rate-limit
const connectsByIp = new Map<string, { count: number; windowStart: number }>()
const CONN_WINDOW_MS = 60_000
const CONN_MAX = 10

export function checkConnectRate(ip: string): boolean {
    const now = Date.now()
    const entry = connectsByIp.get(ip)
    if (!entry || now - entry.windowStart > CONN_WINDOW_MS) {
        connectsByIp.set(ip, { count: 1, windowStart: now })
        return true
    }
    if (entry.count >= CONN_MAX) return false
    entry.count += 1
    return true
}

// DNSBL (Spamhaus Zen)
const dnsblCache = new Map<string, { listed: boolean; at: number }>()
const DNSBL_TTL_MS = 60 * 60 * 1000

export async function isSpamhausListed(ip: string): Promise<boolean> {
    if (!/^\d+\.\d+\.\d+\.\d+$/.test(ip)) return false
    if (ip.startsWith('127.') || ip.startsWith('10.') || ip.startsWith('192.168.') || ip.startsWith('172.16.')) return false
    const cached = dnsblCache.get(ip)
    if (cached && Date.now() - cached.at < DNSBL_TTL_MS) return cached.listed

    const reversed = ip.split('.').reverse().join('.')
    try {
        await dns.resolve4(`${reversed}.zen.spamhaus.org`)
        dnsblCache.set(ip, { listed: true, at: Date.now() })
        return true
    } catch {
        dnsblCache.set(ip, { listed: false, at: Date.now() })
        return false
    }
}

// Greylisting
const greylist = new Map<string, number>()
const GREY_HOLD_MS = 5 * 60 * 1000
const GREY_TTL_MS = 24 * 60 * 60 * 1000

export function shouldGreylist(ip: string, from: string, to: string): boolean {
    const key = `${ip}|${from.toLowerCase()}|${to.toLowerCase()}`
    const first = greylist.get(key)
    const now = Date.now()
    if (!first) {
        greylist.set(key, now)
        return true
    }
    return now - first < GREY_HOLD_MS
}

setInterval(() => {
    const now = Date.now()
    for (const [ip, e] of connectsByIp) if (now - e.windowStart > CONN_WINDOW_MS * 2) connectsByIp.delete(ip)
    for (const [k, t] of greylist) if (now - t > GREY_TTL_MS) greylist.delete(k)
    for (const [ip, e] of dnsblCache) if (now - e.at > DNSBL_TTL_MS) dnsblCache.delete(ip)
}, 60 * 60 * 1000).unref()
```

### 3. Wire into `src/server/mx-server.ts`

```ts
import { checkConnectRate, isSpamhausListed, shouldGreylist } from './lib/mx-guard'

const server = new SMTPServer({
    // ... existing config ...
    maxRecipients: 50,
    size: 25 * 1024 * 1024,

    async onConnect(session, callback) {
        const ip = session.remoteAddress || 'unknown'
        if (!checkConnectRate(ip)) {
            return callback(new Error('421 4.7.0 Too many connections'))
        }
        if (await isSpamhausListed(ip)) {
            console.log(`[MX] Rejected listed IP: ${ip}`)
            return callback(new Error('554 5.7.1 Blacklisted by Spamhaus'))
        }
        callback()
    },

    async onRcptTo(address, session, callback) {
        // ... existing local/routed check ...
        const ip = session.remoteAddress || 'unknown'
        const from = session.envelope.mailFrom?.address || ''
        if (shouldGreylist(ip, from, address.address)) {
            return callback(new Error('451 4.7.1 Greylisted; please retry in 5 minutes'))
        }
        callback()
    },

    // ... existing onData ...
})
```

## Risk
- False positives on forwarding services (mailing lists, aliases) — some legitimate mail from `forwarded@listserv.com` may fail DNSBL or DMARC. Mitigate with allowlist of known relay IPs (future phase).
- Greylisting in-memory Map resets on deploy → every sender re-greylisted. If deploy frequency is >1/day, migrate to DB-backed greylist.
- DNSBL is a third-party dependency (Spamhaus); if they're down, fail-open (don't block).

## Rollback
Comment out the guards in `onConnect`/`onRcptTo`; MX accepts everything again. App keeps working but spam fills INBOX.
