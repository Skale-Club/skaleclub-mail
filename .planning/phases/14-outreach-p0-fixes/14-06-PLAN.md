    ```ts
    // P0-06: Postgres advisory lock — prevents two container instances (or a blue-green deploy
    // overlap) from running the processor concurrently. Replaces the in-memory isSequenceProcessing
    // flag previously in jobs/index.ts which only protected within a single Node process.
    const LOCK_ID_OUTREACH_PROCESSOR = 4014

    export async function runOutreachProcessorWithLock(): Promise<{ acquired: boolean; result?: { processed: number; sent: number; errors: number } }> {
        // pg_try_advisory_lock returns true if we got the lock, false if another connection holds it.
        const lockResult = await db.execute(sql`SELECT pg_try_advisory_lock(${LOCK_ID_OUTREACH_PROCESSOR}) AS acquired`)
        const acquired = (lockResult as unknown as Array<{ acquired: boolean }>)[0]?.acquired === true
        if (!acquired) {
            console.log('[processOutreachSequences] advisory lock held by another instance — skipping tick')
            return { acquired: false }
        }
        try {
            const result = await processOutreachSequences()
            return { acquired: true, result }
        } finally {
            await db.execute(sql`SELECT pg_advisory_unlock(${LOCK_ID_OUTREACH_PROCESSOR})`)
        }
    }
    ```

    Note on the result-shape narrowing: this project uses postgres-js (see `src/db/index.ts:1-2`), whose `db.execute(sql\`...\`)` returns an array of row objects directly (NOT a `{ rows: [...] }` envelope). The cast `(lockResult as unknown as Array<{ acquired: boolean }>)[0]?.acquired === true` is correct for this driver. If you need a defensive idiom that also handles node-pg shape (in case the driver changes), use:
    ```ts
    const acquired = Array.isArray(lockResult)
        ? (lockResult as Array<{ acquired: boolean }>)[0]?.acquired === true
        : (lockResult as { rows: Array<{ acquired: boolean }> }).rows?.[0]?.acquired === true
    ```
